def num_of_like(value):
    if (value <= 10):
        return(3)
    elif (value > 10 and value <=20):
        return(7)
    elif (value >20 and value<=30):
        return(12)
    elif (value >30 and value<=40):
        return(15)
    elif (value >=40):
        return(20)
